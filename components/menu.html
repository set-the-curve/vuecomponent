<template>
  <div class="menu">
    <ul v-for="(group, index) in groups" :key="index">
      <h4 @click="toggleHeader(index)">{{group.title}} <fa v-if="!toggleByHeader[index]" icon="plus" size="xs" /><fa v-if="toggleByHeader[index]" icon="minus" size="xs" /></h4>
        <div v-if="toggleByHeader[index]">
          <a @click.native="maybeHide()" v-for="(item, index2) in group.content" :key="index2" :to="item.path" tag="li" @mouseover.native="showByIndex = index2; showByHeader = index;" @mouseout.native="showByIndex = null; showByHeader = null;">
            <span v-bind:class="{'circle': showByIndex == index2 && showByHeader == index}" class="hidecircle"></span>{{item.name}}
          </a>
        </div>
    </ul>
  </div>
</template>

<script>
import Vue from 'vue'

export default {
  name: 'Menu',
  props: {
  },
  data() {
    return {
      showByIndex: null,
      showByHeader: null,
      toggleByHeader: [],
      groups: [
        {'content':
            [{'name': 'Alert', 'path': '/components/alert'},
            {'name': 'Button', 'path': '/components/button'},
            {'name':'Carousel','path':'/components/carousel'},
            {'name':'Collapse','path':'/components/collapse'},
            /*{'name':'CustomizeBox', 'path':'/components/customize_box'},*/
            {'name': 'Tabs', 'path': '/components/tabs'},],
          'title': 'Action'
        },
        {'content':
            [{'name':'Aside', 'path':'/components/aside'},
            {'name':'Footer', 'path':'/components/footer'},
            {'name':'Navbar', 'path':'/components/navbar'},
            {'name':'Navmenu', 'path':'/components/navmenu'},
            {'name':'StickyRight', 'path':'/components/stickyright'}],
         'title': 'Layouts'
        },
        {'content':
            [{'name':'Autocomplete', 'path':'/components/autocomplete'},
            {'name':'Dropdown', 'path':'/components/dropdown'}],
          'title': 'Dropdown'
        },
        {'content':
            [{'name': 'Avatar', 'path': '/components/avatar'},
            {'name': 'Badge', 'path': '/components/badge'},
            {'name': 'Card', 'path': '/components/card'},
            {'name': 'Code Block', 'path': '/components/code'},
            {'name': 'Standout', 'path': '/components/standout'}],
          'title': 'Visual'
         }
      ],
    }
  },
  mounted() {
    for (var i=0; i<this.groups.length; i++) {
      if (i==0) {
        this.toggleByHeader[i]=true;
      } else {
        this.toggleByHeader[i]=false;
      }
    }
  },
  methods: {
    toggleHeader: function(index) {
      Vue.set(this.toggleByHeader, index, !this.toggleByHeader[index]);
    },
    maybeHide: function() {
      if (window.innerWidth < 900) {
        this.$store.commit('hide');
      }
    }
  }
}
</script>
