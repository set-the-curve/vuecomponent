    <div class="tab-content">
      <div v-for="(name, index) in names" :key="`tab${index}`" v-if="tabIndex==`${index}`">
        <slot :name="`tab${index}`" ></slot>
      </div>
    </div>
  </div>
</template>

<script>

export default {
  name: 'Tabs',
  components: {},
  props: {
    width: String,
    height: String,
    names: Array

      tabIndex: 0

  },
}
</script>
